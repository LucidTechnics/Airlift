127 /**
128  * @author Bediako George
129  * 
130  * @description This method will collect active records for a domain.
131  * It may accept a config object that contains configuration
132  * information for executing the request for the collection.  The
133  * following may be specified in the config object.
134  *
135  * 		<p>offset - Optional. Indicates the start index of the collection.
136  * 		0 is the default value.</p>
137  *
138  * 		<p>limit - Optional. Indicates the maximum amount that should be returned by
139  * 		this query. 10 is the default value.</p>
140  *
141  * 		<p>asc - Optional. Set to true for sorting records in ascending order, false
142  * 		for descending order.  True is the default value.</p>
143  *
144  * 		<p>orderBy - Optional. Name of the property on this domain by
145  * 		which this collection should be sorted.  Set to "auditPutDate"
146  * 		as the default value.</p>
147  *
148  * 		<p>returnType- Optional. Determine the type of object that will
149  * 		hold the collection returned.  Could be "asList" for an
150  * 		enhanced list, "asIterator" for an enhanced iterator, or
151  * 		"asIterable" for an enhanced list.  This is set to "asIterator"
152  * 		by default.</p>
153  * 		
154  * 		<p>filterList - Optional. A list of filter objects that will allow
155  * 		the developer to further refine the results in the collection.
156  * 		A filter object has three required properties, namely ...
157  *
158  *			<p>(i)"attribute" which is the name of the property on a member
159  *			entity of the domain collection,</p>
160  *
161  *			<p>(ii)"operatorName" which is the
162  *			name of the comparison operator to be used to compare this
163  *			attribute as defined by the Java enum
164  *			com.google.appengine.api.datastore.Query.FilterOperator</p>
165  *
166  *			<p>(iii)"value" the value that the attribute value must be
167  *			compared with via the operation specified by
168  *			"operatorName".</p>
169  *		</p>
170  *
171  * @param _arg1 - Optional. If this is a string it will be taken to be the name
172  * of the domain that should be collected.  If it is not a string it
173  * must be a config object.
174  *
175  * @param _arg2 - Optional, If arg1 is defined then _arg2 is the config
176  * object.
177  *
178  * @return - the active record used to collect the domain results.
179  * @return - either an enhanced list, a java.util.Iterable, or an enhanced
180  * java.util.Iterator.  This depends on what was specified in the
181  * config object.
182  *
183  * @example
184  * if the URI is www.example.com/a/runner
185  *
186  * then one could write
187  * 
188  * var [runner, runners] = airlift.collect();
189  *
190  * and this will retrieve at most ten runners ordered by "auditPutDate"
191  * in the runners enhanced iterator.
192  *
193  * To get the runners
194  *
195  * for (var runner in Iterator(runners))
196  * {
197  *		LOG.info(runner.json());
198  * }
199  *
200  * or this works as well
201  *
202  * runners.forEach(function(_runner) { LOG.info(_runner.json()); });
203  *
204  * Also one could do this ...
205  *
206  * var [runner, runnersList] = airlift.collect(
207  * "prorunner",
208  * {
209  *	returnType: "asList",
210  *	orderBy: "firstName",
211  *	limit:200,
212  *	offset:100,
213  *	filterList:[
214  *	{attribute:"lastName", operatorName:"EQUAL", value:"George"},
215  *	{attribute:"age", operatorName:"LESS_THAN" value:20}
216  *	]
217  * };
218  *
219  * and this will return an enhanced list that contains "prorunner"
220  * active records that have last name "George" and age less than 20
221  * years old, paginated from 100 to 200, and ordered by the first name.
222  *
223  * One may get the runners as shown above.
224  *
225  */
226 airlift.collect = function(_arg1, _arg2)
227 {
228 	if (_arg1 && _arg2)
229 	{
230 		//_arg1 is the domain name and arg2 the config object
231 		var domainName = _arg1;
232 		var config = _arg2;
233 	}
234 	else if (_arg1 && airlift.isDefined(_arg1.length) === true)
235 	{
236 		//_arg1 is a string or java.lang.String and _arg2 os not
237 		//defined.
238 		var domainName = _arg1;
239 	}
240 	else if (_arg1 && typeof _arg1 === 'object')
241 	{
242 		//_arg1 is a config object
243 		var domainName = DOMAIN_NAME;
244 		var config = _arg1;
245 	}
246 	else
247 	{
248 		//nothing was passed in ...
249 		var domainName = DOMAIN_NAME;
250 		var config = {};
251 	}
252 
253 	config.limit = config.limit||REQUEST.getParameter("limit")||10;
254 	config.offset = config.offset||REQUEST.getParameter("offset")||0;
255 	config.orderBy = config.orderBy||REQUEST.getParameter("orderBy")||"auditPutDate";
256 	
257 	function evaluateBooleanString(_booleanString)
258 	{
259 		if ("true".equalsIgnoreCase(_booleanString) === true)
260 		{
261 			var value = true;
262 		}
263 		else
264 		{
265 			var value = false;

267 
268 		return value;
269 	}
270 	
271 	config.asc = (airlift.isDefined(config.asc) === true) ? config.asc : (airlift.isDefined(REQUEST.getParameter("asc")) === true) ? evaluateBooleanString(REQUEST.getParameter("asc")) : false;
272 	
273 	var activeRecord = airlift.ar(domainName);
274 	var result = activeRecord.collect(config);
275 
276 	return [activeRecord, result];
